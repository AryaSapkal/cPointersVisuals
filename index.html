Processes in C

#include <stdio.h>

#include <unistd.h>

int sum(int nums[], int arraySize){

 int sum = 0;

 for(int i = 0; i < arraySize; i++){

  sum += nums[i];

 }

 return sum;

}

int main(){

 int numbers1[] = {1, 2, 3, 4};

 int numbers2[] = {5, 6, 7, 8};

 int arraySize = 4;

 int forkValue = fork();

 if(forkValue == 0){

  printf("Child's work: %d\n", sum(numbers1, arraySize));

 }

 else{

  printf("Parent's work: %d\n", sum(numbers2, arraySize));

 }

}
